\begin{soal}[Chapra Latihan 15.3]
Selesaikan permasalahan pemrograman linear berikut.
Cari nilai maksimum dari:
\begin{equation*}
f(x,y) = 1.75x + 1.25y
\end{equation*}
dengan kendala:
\begin{align*}
1.2x + 2.25y \leq 14 \\
x + 1.1y \leq 8 \\
2.5x + y \leq 9
\end{align*}
serta batas $x \geq 0$ dan $y \geq 0$.
\end{soal}


\begin{soal}[Chapra Latihan 15.4]
Selesaikan permasalahan pemrograman linear berikut.
Cari nilai maksimum dari:
\begin{equation*}
f(x,y) = 6x + 8y
\end{equation*}
dengan kendala:
\begin{align*}
5x + 2y \leq 40 \\
6x + 6y \leq 60 \\
2x + 4y \leq 32
\end{align*}
serta batas $x \geq 0$ dan $y \geq 0$.
\end{soal}
